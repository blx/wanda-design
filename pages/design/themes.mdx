import { useState } from "react";
import { MdxLayout } from "@/components/layouts/mdx";
import { ThemeList } from "@/components/theme-list";
import lightTheme from "@wonderflow/react-components/themes/light.json";
import darkTheme from "@wonderflow/react-components/themes/dark.json";
import { Stack, Radio, Text, Separator } from "@wonderflow/react-components";
import { UIDReset, UIDConsumer } from "react-uid";

export default ({ children }) => (
  <MdxLayout title="Themes" description="Color schemes for scalability">
    {children}
  </MdxLayout>
);

Wanda introduces a well-proven theme system which help designers and developer building a product.

A theme is a set of predefined `key â†” color` combinations to be used whenever possible in order keep consistency, accessibility and dynamicity.

## Available themes

Here you can check the theme keys for all the available themes from Wanda. These themes are shared as library on Figma and as `css/json` from the [components library](/develop/using-themes/).

export const ThemesSwitcher = () => {
  const [theme, setTheme] = useState(lightTheme);
  return (
    <UIDReset>
      <UIDConsumer>
        {(id, uid) => (
          <Stack rowGap={32}>
            <Separator aria-hidden="true" />
            <Stack
              direction="row"
              horizontalAlign="center"
              fill={false}
              columnGap={24}
              inline
            >
              <Stack verticalAlign="center" direction="row" columnGap={8}>
                <Radio
                  dimension="small"
                  id={uid("light")}
                  onChange={() => setTheme(lightTheme)}
                  name="chooseTheme"
                  value="LightTheme"
                  defaultChecked
                />
                <Text as="label" htmlFor={uid("light")}>
                  <b>Light</b>
                </Text>
              </Stack>
              <Stack verticalAlign="center" direction="row" columnGap={8}>
                <Radio
                  dimension="small"
                  id={uid("dark")}
                  onChange={() => setTheme(darkTheme)}
                  name="chooseTheme"
                  value="DarkTheme"
                />
                <Text as="label" htmlFor={uid("dark")}>
                  <b>Dark</b>
                </Text>
              </Stack>
            </Stack>
            <Separator aria-hidden="true" />
            <ThemeList theme={theme} />
          </Stack>
        )}
      </UIDConsumer>
    </UIDReset>
  );
};

<ThemesSwitcher />
